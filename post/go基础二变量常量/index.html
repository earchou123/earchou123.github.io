go基础（二）变量&常量

Tue, Mar 7, 2023

变量

变量声明

语法

var 变量名字 类型 = 表达式

var name string

批量声明

var(
    name string
    age int
    b bool
)

变量初始化

var name string = "hahah"

类型推断

var name = "haha"
var age = 18

批量声明

var name,age,b = "haha",123,true

短变量声明
只能放在函数内部，不能放在函数外部

age := 123

常量

常量定义

const Name [type] = value

const PI float32 = 3.14

常量一旦被定义，无法被修改

iota

可被修改的常量编辑器

const (
    a1 = iota   //0
    a2 = iota   //1
    a3 = iota   //2
)

跳过某个值

const (
    a1 = iota   //0
    _           //1
    a3 = iota   //2
)
